167 void Atlas::onHashQuery(uint32_t blockNumber, uint8_t mapNumber)
168 {
169  Logger::g_pLogger->LogPrint("Atlas: Got Hash Query\n");
170  uint16_t* crcs = GetGroupOfBlockCrcs(mapNumber, blockNumber);
171  uint8_t* pResponse = new uint8_t[71];
172 
173  pResponse[0] = 0x3F; //byte 000 - cmd
174  *reinterpret_cast<uint16_t*>(pResponse + 1) = 71; //byte 001 through 002 - packet size
175  *reinterpret_cast<uint32_t*>(pResponse + 3) = htonl(blockNumber); //byte 003 through 006 - central block number for the query (block that player is standing in)
176  *reinterpret_cast<uint32_t*>(pResponse + 7) = htonl(8); //byte 007 through 010 - number of statics in the packet (8 for a query response)
177  *reinterpret_cast<uint16_t*>(pResponse + 11) = htons(0); //byte 011 through 012 - UltimaLive sequence number
178  pResponse[13] = 0xFF; //byte 013 - UltimaLive command (0xFF is a block Query Response)
179  pResponse[14] = mapNumber; //byte 014 - UltimaLive mapnumber
180  //byte 015 through 64 - 25 block CRCs
181  for (int i = 0; i < 25; i++)
182  {
183  *reinterpret_cast<uint16_t*>(pResponse + 15 + (i * 2)) = htons(crcs[i]);
184  }
185 
186  pResponse[65] = 0xFF; //byte 065 - padding
187  pResponse[66] = 0xFF; //byte 066 - padding
188  pResponse[67] = 0xFF; //byte 067 - padding
189  pResponse[68] = 0xFF; //byte 068 - padding
190  pResponse[69] = 0xFF; //byte 069 - padding
191  pResponse[70] = 0xFF; //byte 070 - padding
192 
193  m_pNetworkManager->sendPacketToServer(pResponse);
194 
195  delete pResponse;
196 }

255 uint16_t Atlas::fletcher16(uint8_t* pBlockData, uint8_t* pStaticsData, uint32_t staticsLength)
256 {
257  uint16_t sum1 = 0;
258  uint16_t sum2 = 0;
259 
260  if (pBlockData != NULL)
261  {
262  for (int index = 0; index < 192; ++index)
263  {
264  sum1 = (uint16_t)((sum1 + pBlockData[index]) % 255);
265  sum2 = (uint16_t)((sum2 + sum1) % 255);
266  }
267  }
268 
269  if (pStaticsData != NULL)
270  {
271  for (uint32_t index = 0; index < staticsLength; ++index)
272  {
273  sum1 = (uint16_t)((sum1 + pStaticsData[index]) % 255);
274  sum2 = (uint16_t)((sum2 + sum1) % 255);
275  }
276  }
277 
278  return (uint16_t)((sum2 << 8) | sum1);
279 }

283 uint16_t* Atlas::GetGroupOfBlockCrcs(uint32_t mapNumber, uint32_t blockNumber)
284 {
285  uint16_t* pCrcs = NULL;
286  if (m_mapDefinitions.find(mapNumber) != m_mapDefinitions.end())
287  {
288  pCrcs = new uint16_t[25];
289  MapDefinition def = m_mapDefinitions[mapNumber];
290  int32_t blockX = ((int32_t)blockNumber / (def.mapHeightInTiles >> 3));
291  int32_t blockY = ((int32_t)blockNumber % (def.mapHeightInTiles >> 3));
292 
293  int32_t wrapWidthInBlocks = (def.mapWrapWidthInTiles >> 3);
294  int32_t wrapHeightInBlocks = (def.mapWrapHeightInTiles >> 3);
295  int32_t mapWidthInBlocks = (def.mapWidthInTiles >> 3);
296  int32_t mapHeightInBlocks = (def.mapHeightInTiles >> 3);
297 
298  if (blockX < 0 || blockX >= mapWidthInBlocks|| blockY < 0 || blockY >= mapHeightInBlocks)
299  {
300  memset(pCrcs, 0x00, sizeof(uint16_t) * 25);
301  }
302  else
303  {
304  for (int x = -2; x <= 2; x++)
305  {
306  int xBlockItr = -1;
307  if (blockX < wrapWidthInBlocks)
308  {
309  xBlockItr = (blockX + x) % wrapWidthInBlocks;
310  if (xBlockItr < 0 && xBlockItr > -3)
311  {
312  xBlockItr += wrapWidthInBlocks;
313  }
314  }
315  else
316  {
317  xBlockItr = (blockX + x) % mapWidthInBlocks;
318  if (xBlockItr < 0 && xBlockItr > -3)
319  {
320  xBlockItr += mapWidthInBlocks;
321  }
322  }
323  for (int y = -2; y <= 2; y++)
324  {
325  int yBlockItr = 0;
326  if (blockY < wrapHeightInBlocks)
327  {
328  yBlockItr = (blockY + y) % wrapHeightInBlocks;
329  if (yBlockItr < 0)
330  {
331  yBlockItr += wrapHeightInBlocks;
332  }
333  }
334  else
335  {
336  yBlockItr = (blockY + y) % mapHeightInBlocks;
337  if (yBlockItr < 0)
338  {
339  yBlockItr += mapHeightInBlocks;
340  }
341  }
342 
343  int32_t currentBlock = (xBlockItr * mapHeightInBlocks) + yBlockItr;
344 
345  if (currentBlock >= 0 && currentBlock <= (mapHeightInBlocks * mapWidthInBlocks))
346  {
347  pCrcs[((x + 2) * 5) + (y + 2)] = getBlockCrc (mapNumber, currentBlock);
348  }
349  else
350  {
351  pCrcs[((x + 2) * 5) + (y + 2)] = (uint16_t)0x0;
352  }
353  }
354  }
355  }
356  }
357 
358  return pCrcs;
359 }

361 uint16_t Atlas::getBlockCrc(uint32_t mapNumber, uint32_t blockNumber)
362 {
363  uint16_t crc = 0;
364  if (m_mapDefinitions.find(mapNumber) != m_mapDefinitions.end())
365  {
366  uint8_t* pBlockData = m_pFileManager->readLandBlock(static_cast<uint8_t>(mapNumber), blockNumber);
367  uint32_t staticsLength = 0;
368 
369  uint8_t* pStaticsData = m_pFileManager->readStaticsBlock(mapNumber, blockNumber, staticsLength);
370 
371  if (pBlockData != NULL)
372  {
373  crc = fletcher16(pBlockData, pStaticsData, staticsLength);
374  delete pBlockData;
375  }
376 
377  if (pStaticsData != NULL)
378  {
379  delete pStaticsData;
380  }
381  }
382 
383  return crc;
384 }
