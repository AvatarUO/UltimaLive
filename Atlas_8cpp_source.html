48 bool Atlas::Initialize()
49 {
50  bool success = true;
51 
52  Logger::g_pLogger->LogPrint("Initializing Atlas\n");
53  Atlas* pInstance = static_cast<Atlas*>(UltimaLive::g_pUltimaLive->Lookup("Atlas"));
54 
55  if (pInstance != NULL)
56  {
57  BaseFileManager* pFileManager = static_cast<BaseFileManager*>(UltimaLive::g_pUltimaLive->Lookup("FileManager"));
58  if (pFileManager == NULL)
59  {
60  Logger::g_pLogger->LogPrintError("Atlas: Failed to acquire File Manager Instance\n");
61  success = false;
62  }
63 
64  Client* pClient = static_cast<Client*>(UltimaLive::g_pUltimaLive->Lookup("Client"));
65  if (pClient == NULL)
66  {
67  Logger::g_pLogger->LogPrintError("Atlas: Failed to acquire Client class Instance\n");
68  success = false;
69  }
70 
71  NetworkManager* pNetworkManager = static_cast<NetworkManager*>(UltimaLive::g_pUltimaLive->Lookup("NetworkManager"));
72  if (pNetworkManager == NULL)
73  {
74  Logger::g_pLogger->LogPrintError("Atlas: Failed to acquire Network Manager Instance\n");
75  success = false;
76  }
77  else
78  {
79  pNetworkManager->subscribeToOnBeforeMapChange(std::bind(&Atlas::onBeforeMapChange, pInstance, std::placeholders::_1));
80  pNetworkManager->subscribeToOnMapChange(std::bind(&Atlas::onMapChange, pInstance, std::placeholders::_1));
81  pNetworkManager->subscribeToRefreshClient(std::bind(&Atlas::onRefreshClientView, pInstance));
82  pNetworkManager->subscribeToBlockQueryRequest(std::bind(&Atlas::onHashQuery, pInstance, std::placeholders::_1, std::placeholders::_2));
83  pNetworkManager->subscribeToStaticsUpdate(std::bind(&Atlas::onUpdateStatics, pInstance, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4));
84  pNetworkManager->subscribeToMapDefinitionUpdate(std::bind(&Atlas::onUpdateMapDefinitions, pInstance, std::placeholders::_1));
85  pNetworkManager->subscribeToLandUpdate(std::bind(&Atlas::onUpdateLand, pInstance, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3));
86  pNetworkManager->subscribeToUltimaLiveLoginComplete(std::bind(&Atlas::onShardIdentifierUpdate, pInstance, std::placeholders::_1));
87  pNetworkManager->subscribeToLogout(std::bind(&Atlas::onLogout, pInstance));
88  }
89 
90  pInstance->m_pFileManager = pFileManager;
91  pInstance->m_pNetworkManager = pNetworkManager;
92  pInstance->m_pClient = pClient;
93  }
94  else
95  {
96  Logger::g_pLogger->LogPrintError("Atlas: Failed to acquire Atlas Instance\n");
97  success = false;
98  }
99 
100  return success;
101 }

212 void Atlas::onHashQuery(uint32_t blockNumber, uint8_t mapNumber)
213 {
214  Logger::g_pLogger->LogPrint("Atlas: Got Hash Query\n");
215  uint16_t* crcs = GetGroupOfBlockCrcs(mapNumber, blockNumber);
216  uint8_t* pResponse = new uint8_t[71];
217 
218  pResponse[0] = 0x3F; //byte 000 - cmd
219  *reinterpret_cast<uint16_t*>(pResponse + 1) = 71; //byte 001 through 002 - packet size
220  *reinterpret_cast<uint32_t*>(pResponse + 3) = htonl(blockNumber); //byte 003 through 006 - central block number for the query (block that player is standing in)
221  *reinterpret_cast<uint32_t*>(pResponse + 7) = htonl(8); //byte 007 through 010 - number of statics in the packet (8 for a query response)
222  *reinterpret_cast<uint16_t*>(pResponse + 11) = htons(0); //byte 011 through 012 - UltimaLive sequence number
223  pResponse[13] = 0xFF; //byte 013 - UltimaLive command (0xFF is a block Query Response)
224  pResponse[14] = mapNumber; //byte 014 - UltimaLive mapnumber
225  //byte 015 through 64 - 25 block CRCs
226  for (int i = 0; i < 25; i++)
227  {
228  *reinterpret_cast<uint16_t*>(pResponse + 15 + (i * 2)) = htons(crcs[i]);
229  }
230 
231  pResponse[65] = 0xFF; //byte 065 - padding
232  pResponse[66] = 0xFF; //byte 066 - padding
233  pResponse[67] = 0xFF; //byte 067 - padding
234  pResponse[68] = 0xFF; //byte 068 - padding
235  pResponse[69] = 0xFF; //byte 069 - padding
236  pResponse[70] = 0xFF; //byte 070 - padding
237 
238  m_pNetworkManager->sendPacketToServer(pResponse);
239 
240  delete pResponse;
241 }

331 uint16_t Atlas::fletcher16(uint8_t* pBlockData, uint8_t* pStaticsData, uint32_t staticsLength)
332 {
333  uint16_t sum1 = 0;
334  uint16_t sum2 = 0;
335 
336  if (pBlockData != NULL)
337  {
338  for (int index = 0; index < 192; ++index)
339  {
340  sum1 = (uint16_t)((sum1 + pBlockData[index]) % 255);
341  sum2 = (uint16_t)((sum2 + sum1) % 255);
342  }
343  }
344 
345  if (pStaticsData != NULL)
346  {
347  for (uint32_t index = 0; index < staticsLength; ++index)
348  {
349  sum1 = (uint16_t)((sum1 + pStaticsData[index]) % 255);
350  sum2 = (uint16_t)((sum2 + sum1) % 255);
351  }
352  }
353 
354  return (uint16_t)((sum2 << 8) | sum1);
355 }

367 uint16_t* Atlas::GetGroupOfBlockCrcs(uint32_t mapNumber, uint32_t blockNumber)
368 {
369  uint16_t* pCrcs = NULL;
370  if (m_mapDefinitions.find(mapNumber) != m_mapDefinitions.end())
371  {
372  pCrcs = new uint16_t[25];
373  MapDefinition def = m_mapDefinitions[mapNumber];
374  int32_t blockX = ((int32_t)blockNumber / (def.mapHeightInTiles >> 3));
375  int32_t blockY = ((int32_t)blockNumber % (def.mapHeightInTiles >> 3));
376 
377  int32_t wrapWidthInBlocks = (def.mapWrapWidthInTiles >> 3);
378  int32_t wrapHeightInBlocks = (def.mapWrapHeightInTiles >> 3);
379  int32_t mapWidthInBlocks = (def.mapWidthInTiles >> 3);
380  int32_t mapHeightInBlocks = (def.mapHeightInTiles >> 3);
381 
382  if (blockX < 0 || blockX >= mapWidthInBlocks|| blockY < 0 || blockY >= mapHeightInBlocks)
383  {
384  memset(pCrcs, 0x00, sizeof(uint16_t) * 25);
385  }
386  else
387  {
388  for (int x = -2; x <= 2; x++)
389  {
390  int xBlockItr = -1;
391  if (blockX < wrapWidthInBlocks)
392  {
393  xBlockItr = (blockX + x) % wrapWidthInBlocks;
394  if (xBlockItr < 0 && xBlockItr > -3)
395  {
396  xBlockItr += wrapWidthInBlocks;
397  }
398  }
399  else
400  {
401  xBlockItr = (blockX + x) % mapWidthInBlocks;
402  if (xBlockItr < 0 && xBlockItr > -3)
403  {
404  xBlockItr += mapWidthInBlocks;
405  }
406  }
407  for (int y = -2; y <= 2; y++)
408  {
409  int yBlockItr = 0;
410  if (blockY < wrapHeightInBlocks)
411  {
412  yBlockItr = (blockY + y) % wrapHeightInBlocks;
413  if (yBlockItr < 0)
414  {
415  yBlockItr += wrapHeightInBlocks;
416  }
417  }
418  else
419  {
420  yBlockItr = (blockY + y) % mapHeightInBlocks;
421  if (yBlockItr < 0)
422  {
423  yBlockItr += mapHeightInBlocks;
424  }
425  }
426 
427  int32_t currentBlock = (xBlockItr * mapHeightInBlocks) + yBlockItr;
428 
429  if (currentBlock >= 0 && currentBlock <= (mapHeightInBlocks * mapWidthInBlocks))
430  {
431  pCrcs[((x + 2) * 5) + (y + 2)] = getBlockCrc (mapNumber, currentBlock);
432  }
433  else
434  {
435  pCrcs[((x + 2) * 5) + (y + 2)] = (uint16_t)0x0;
436  }
437  }
438  }
439  }
440  }
441 
442  return pCrcs;
443 }

451 uint16_t Atlas::getBlockCrc(uint32_t mapNumber, uint32_t blockNumber)
452 {
453  uint16_t crc = 0;
454  if (m_mapDefinitions.find(mapNumber) != m_mapDefinitions.end())
455  {
456  uint8_t* pBlockData = m_pFileManager->readLandBlock(static_cast<uint8_t>(mapNumber), blockNumber);
457  uint32_t staticsLength = 0;
458 
459  uint8_t* pStaticsData = m_pFileManager->readStaticsBlock(mapNumber, blockNumber, staticsLength);
460 
461  if (pBlockData != NULL)
462  {
463  crc = fletcher16(pBlockData, pStaticsData, staticsLength);
464  delete pBlockData;
465  }
466 
467  if (pStaticsData != NULL)
468  {
469  delete pStaticsData;
470  }
471  }
472 
473  return crc;
474 }
