409 void BaseFileManager::copyFile(std::string sourceFilePath, std::string destFilePath, ProgressBarDialog* pProgress)
410 {
411  std::ifstream sourceFile(sourceFilePath, std::ifstream::binary);
412 
413  if (!sourceFile.good())
414  {
415  sourceFile.close();
416  return;
417  }
418 
419  sourceFile.seekg(0, sourceFile.end);
420  std::streamoff totalFileSizeInBytes = sourceFile.tellg();
421  sourceFile.close();
422 
423  uint32_t bytesCopied = 0;
424  uint32_t prevPercent = 0;
425 
426  // allocate memory for buffer
427  char* buffer = new char[4096];
428  size_t bytesToWrite;
429 
430  FILE* pSource = NULL;
431  FILE* pDest = NULL;
432 
433  errno_t err = fopen_s(&pSource, sourceFilePath.c_str(), "rb");
434  err = fopen_s(&pDest, destFilePath.c_str(), "wb");
435 
436  // clean and more secure
437  // feof(FILE* stream) returns non-zero if the end of file indicator for stream is set
438  do
439  {
440  bytesToWrite = fread(buffer, 1, 4096, pSource);
441  if (bytesToWrite != 0)
442  {
443  fwrite(buffer, 1, bytesToWrite, pDest);
444  bytesCopied += bytesToWrite;
445 
446  if (pProgress != NULL)
447  {
448  uint32_t percent = (uint32_t)(((float)bytesCopied / (float)totalFileSizeInBytes) * 100.0f);
449  if ((uint32_t)percent > prevPercent)
450  {
451  prevPercent = percent;
452  pProgress->setProgress((uint32_t)percent);
453  }
454  }
455  }
456  } while (bytesToWrite != 0);
457 
458  fclose(pSource);
459  fclose(pDest);
