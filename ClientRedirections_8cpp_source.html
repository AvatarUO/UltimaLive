33 bool ClientRedirections::InstallClientHooks()
34 {
35  bool success = true;
36 
37  Logger::g_pLogger->LogPrint("Proceeding to install UltimaLive client hooks...\n");
38  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Initializing function redirector");
39 
40  LocalPeHelper32 ultimaHelper(Utils::GetCurrentModuleName());
41  if (ultimaHelper.Init())
42  {
43  Logger::g_pLogger->LogPrintTaskStatusResult(true);
44  }
45  else
46  {
47  Logger::g_pLogger->LogPrintTaskStatusResult(false);
48  Logger::g_pLogger->LogPrint("Unable to initialize function redirector! Fatal!");
49  success = false;
50  }
51 
52  if (success)
53  {
54  //remap CreateFileA to avoid calling this multiple times
55  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Redirecting CreateFileA");
56  if (!ultimaHelper.PatchImportedFunctionAddress("kernel32.dll", "CreateFileA", (DWORD)&ClientRedirections::OnCreateFileA))
57  {
58  Logger::g_pLogger->LogPrintTaskStatusResult(false);
59  Logger::g_pLogger->LogPrint("Failed to hook OnCreateFileA\n");
60  }
61  else
62  {
63  Logger::g_pLogger->LogPrintTaskStatusResult(true);
64  }
65 
66  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Redirecting MessageBoxA");
67  if (!ultimaHelper.PatchImportedFunctionAddress("user32.dll", "MessageBoxA", (DWORD)&ClientRedirections::OnMessageBox))
68  {
69  Logger::g_pLogger->LogPrintTaskStatusResult(false);
70  Logger::g_pLogger->LogPrint("Failed to hook MessageBox\n");
71  }
72  else
73  {
74  Logger::g_pLogger->LogPrintTaskStatusResult(true);
75  }
76 
77  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Redirecting CreateFileMappingA");
78  if (!ultimaHelper.PatchImportedFunctionAddress("kernel32.dll", "CreateFileMappingA", (DWORD)&ClientRedirections::OnCreateFileMappingA))
79  {
80  Logger::g_pLogger->LogPrintTaskStatusResult(false);
81  Logger::g_pLogger->LogPrint("Failed to hook CreateFileMappingA\n");
82  }
83  else
84  {
85  Logger::g_pLogger->LogPrintTaskStatusResult(true);
86  }
87 
88  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Redirecting CloseHandle");
89  if (!ultimaHelper.PatchImportedFunctionAddress("kernel32.dll", "CloseHandle", (DWORD)&ClientRedirections::OnCloseHandle))
90  {
91  Logger::g_pLogger->LogPrintTaskStatusResult(false);
92  Logger::g_pLogger->LogPrint("Failed to hook CloseHandle\n");
93  }
94  else
95  {
96  Logger::g_pLogger->LogPrintTaskStatusResult(true);
97  }
98 
99  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Redirecting MapViewOfFile");
100  if (!ultimaHelper.PatchImportedFunctionAddress("kernel32.dll", "MapViewOfFile", (DWORD)&ClientRedirections::OnMapViewOfFile))
101  {
102  Logger::g_pLogger->LogPrintTaskStatusResult(false);
103  Logger::g_pLogger->LogPrint("Failed to hook CreateFileMappingA\n");
104  }
105  else
106  {
107  Logger::g_pLogger->LogPrintTaskStatusResult(true);
108  }
109 
110  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Hooking client send function");
111  BOOL bHook = Mhook_SetHook((PVOID*)&UltimaLive::g_pUltimaLive->m_pClient->m_pSendFunction, ClientRedirections::OnSendPacket);
112  if (!bHook)
113  {
114  Logger::g_pLogger->LogPrintTaskStatusResult(false);
115  success = false;
116  }
117  else
118  {
119  Logger::g_pLogger->LogPrintTaskStatusResult(true);
120  }
121 
122  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Hooking client receive function");
123  bHook = Mhook_SetHook((PVOID*)&UltimaLive::g_pUltimaLive->m_pClient->m_pRecvFunction, ClientRedirections::OnReceivePacket);
124  if (!bHook)
125  {
126  Logger::g_pLogger->LogPrintTaskStatusResult(false);
127  success = false;
128  }
129  else
130  {
131  Logger::g_pLogger->LogPrintTaskStatusResult(true);
132  }
133 
134  Logger::g_pLogger->LogPrintWithoutDate("%-87s", " Hooking client update blocks function");
135  bHook = Mhook_SetHook((PVOID*)&UltimaLive::g_pUltimaLive->m_pClient->m_pUpdateBlocksFunction, ClientRedirections::OnUpdateStaticBlocks);
136  if (!bHook)
137  {
138  Logger::g_pLogger->LogPrintTaskStatusResult(false);
139  success = false;
140  }
141  else
142  {
143  Logger::g_pLogger->LogPrintTaskStatusResult(true);
144  }
145  }
146  return success;
147 }
