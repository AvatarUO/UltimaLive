152 /* @brief This function is called in place of the client's network receive function. It
153  * passes control to the UltimaLive network manager which gives its subscribers the chance
154  * to alter/process each network packet before the client. It then calls the clients
155  * receive function.
156  */
157 void __fastcall ClientRedirections::OnReceivePacket(void* This, void*, unsigned char* pBuffer)
158 {
159  if (UltimaLive::g_pUltimaLive->m_pClient->GetNetworkManager()->OnReceivePacket(pBuffer) == true)
160  {
161  UltimaLive::g_pUltimaLive->m_pClient->getRecvFunction()(This, pBuffer);
162  }
163 }
164 
165 /* @brief This function is called in place of the client's network send function. It
166 * passes control to the UltimaLive network manager which gives its subscribers the chance
167 * to alter/process each network packet before they are sent to the server. It then calls
168 * the clients receive function.
169 */
170 int __fastcall ClientRedirections::OnSendPacket(void* This, void*, unsigned char* pBuffer)
171 {
172  if (UltimaLive::g_pUltimaLive->m_pClient->GetNetworkManager()->OnSendPacket(pBuffer) == true)
173  {
174  int retValue = UltimaLive::g_pUltimaLive->m_pClient->getSendFunction()(This, pBuffer);
175 
176  return retValue;
177  }
178 
179  return 0;
180 }

182 /* @brief This function adds a mutex around the updates static blocks function in the

186 void ClientRedirections::OnUpdateStaticBlocks()
