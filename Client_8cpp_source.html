56 
57  Client* pInstance = static_cast<Client*>(UltimaLive::g_pUltimaLive->Lookup("Client"));
58 
59  if (pInstance != NULL)

61  NetworkManager* pNetworkManager = static_cast<NetworkManager*>(UltimaLive::g_pUltimaLive->Lookup("NetworkManager"));
62  if (pNetworkManager != NULL)
63  {
64  pInstance->m_pNetworkManager = pNetworkManager;
65 
66  BaseFileManager* pFileManager = static_cast<BaseFileManager*>(UltimaLive::g_pUltimaLive->Lookup("FileManager"));
67  if (pFileManager != NULL)

69  pInstance->m_pFileManager = pFileManager;
70  if (!pInstance->LookupClientInternalStructuresAndFunctions())
71  {
72  success = false;
73  Logger::g_pLogger->LogPrintError("FAILED to find all internal client structures and pointers.\n");
74  }
75  else
76  {
77  Logger::g_pLogger->LogPrint("All internal client structures and pointers found.\n");
78  }
79  }
80  else
81  {
82  Logger::g_pLogger->LogPrintError("Client: Failed to acquire FileManager Instance\n");

84  }
85 
86  }
87  else
88  {
89  Logger::g_pLogger->LogPrintError("Client: Failed to acquire NetworkManager Instance\n");
90  success = false;
91  }
92  }
93  else
94  {
95  Logger::g_pLogger->LogPrintError("Client: Failed to acquire Client Instance\n");

489 void Client::refreshClientStatics(uint32_t blockNumber)
490 {
491  Logger::g_pLogger->LogPrint("Refreshing Client View: %i\n", blockNumber);
492 
493  //PURGE MASTER STATIC LIST
494  for (CStaticObject* pStaticItem = *(CStaticObject**)m_pMasterStaticsList; pStaticItem != NULL && pStaticItem->drawItemMembers.bitPattern == 0xFEEDBEEF; pStaticItem = pStaticItem->staticObjectMembers.pNextStaticObject)
495  {
496  uint32_t staticBlockNumber = ((pStaticItem->drawItemMembers.x >> 3) * (m_pMapDimensionsStructure->mapHeightInTiles >> 3)) + (pStaticItem->drawItemMembers.y >> 3);
497  if (staticBlockNumber == blockNumber && !pStaticItem->pVtable->drawItemVtableMembers.IsDynamic(pStaticItem))
498  {
499  if (pStaticItem->drawItemMembers.inDrawList != 0)
500  {
501  pStaticItem->pVtable->drawItemVtableMembers.RemoveFromDrawList(pStaticItem);
502  }
503  }
504  }
505 
506  //Clear Client Blocks Array
507  for (int i = 0; i < 36; i++)
508  {
509  if ( reinterpret_cast<uint32_t*>(m_pClientDisplayedBlocksTable)[i] == blockNumber)
510  {
511  reinterpret_cast<int*>(m_pClientDisplayedBlocksTable)[i] = -1;
512  }
513  }
514 
515  //Set Client Minimum Displayed X to -9999
516  m_pMapMinCoordDisplayStructure->minClientDisplayX = static_cast<uint32_t>(-9999);
517  m_pMapMinCoordDisplayStructure->minClientDisplayY = static_cast<uint32_t>(-9999);
518 
519  ClientRedirections::OnUpdateStaticBlocks();
520 }
