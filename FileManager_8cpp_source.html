183  __in HANDLE hFileMappingObject,
184  __in DWORD dwDesiredAccess,
185  __in DWORD dwFileOffsetHigh,
186  __in DWORD dwFileOffsetLow,
187  __in SIZE_T dwNumberOfBytesToMap
188  )
189 {
190  ClientFileHandleSet* pMatchingFileset = NULL;
191 
192  for (std::map<std::string, ClientFileHandleSet*>::const_iterator iterator = m_files.begin(); iterator != m_files.end(); iterator++)
193  {
194  if (iterator->second->m_createFileMappingHandle == hFileMappingObject)
195  {
196  pMatchingFileset = iterator->second;
197  break;
198  }
199  }
200 
201  HANDLE handleToReturn = INVALID_HANDLE_VALUE;
202 
203  if (pMatchingFileset != NULL)
204  {
205  std::string shortFilename = Utils::getFilenameFromPath(pMatchingFileset->m_filename);
206  std::map<std::string, uint32_t>::iterator iterator = m_neededFiles.find(shortFilename);
207 
208  if (iterator != m_neededFiles.end())
209  {
210  if (shortFilename.find("map") != std::string::npos)
211  {
212  handleToReturn = m_pMapPool;
213  }
214  else if (shortFilename.find("statics") != std::string::npos)
215  {
216  handleToReturn = m_pStaticsPool;
217  }
218  else if (shortFilename.find("staidx") != std::string::npos)
219  {
220  handleToReturn = m_pStaidxPool;
221  }
222  }
223  }
224 
225  if (handleToReturn == INVALID_HANDLE_VALUE)
226  {
227  handleToReturn = MapViewOfFile(hFileMappingObject, dwDesiredAccess, dwFileOffsetHigh, dwFileOffsetLow, dwNumberOfBytesToMap);
228  if (pMatchingFileset != NULL)
229  {
230  pMatchingFileset->m_mapFileViewHandle = handleToReturn;
231  }
232  }
233 
234  return handleToReturn;
235 }
