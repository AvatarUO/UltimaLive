68 void FileManager::LoadMap(uint8_t mapNumber)
69 {
70  if (m_pMapFileStream->is_open())
71  {
72  m_pMapFileStream->close();
73  }
74  if (m_pStaidxFileStream->is_open())
75  {
76  m_pStaidxFileStream->close();
77  }
78  if (m_pStaticsFileStream->is_open())
79  {
80  m_pStaticsFileStream->close();
81  }
82 
83  std::string filenameAndPath = BaseFileManager::getUltimaLiveSavePath();
84  if (m_shardIdentifier != "")
85  {
86  filenameAndPath.append(m_shardIdentifier);
87  filenameAndPath.append("\\");
88  }
89 
90  std::string mapFileNameAndPath(filenameAndPath);
91  char filename[32];
92  sprintf_s(filename, "map%i.mul", mapNumber);
93  mapFileNameAndPath.append(filename);
94 
95  std::string staidxFileNameAndPath(filenameAndPath);
96  sprintf_s(filename, "staidx%i.mul", mapNumber);
97  staidxFileNameAndPath.append(filename);
98 
99  std::string staticsFileNameAndPath(filenameAndPath);
100  sprintf_s(filename, "statics%i.mul", mapNumber);
101  staticsFileNameAndPath.append(filename);
102 
103  Logger::g_pLogger->LogPrint("Loading Map: %s\n", mapFileNameAndPath.c_str());
104 
105  std::ifstream mapFile;
106  mapFile.open(mapFileNameAndPath, std::ios::binary | std::ios::in);
107  if (mapFile.is_open())
108  {
109  mapFile.seekg (0, mapFile.end);
110  std::streamoff length = mapFile.tellg();
111  if (length > 0)
112  {
113  mapFile.seekg (0, mapFile.beg);
114  mapFile.read(reinterpret_cast<char*>(m_pMapPool), length);
115  }
116 
117  mapFile.close();
118  }
119 
120  Logger::g_pLogger->LogPrint("Loading Staidx: %s\n", staidxFileNameAndPath.c_str());
121 
122  std::ifstream staidxFile;
123  staidxFile.open(staidxFileNameAndPath, std::ios::binary | std::ios::in);
124 
125  if (staidxFile.is_open())
126  {
127  staidxFile.seekg (0, staidxFile.end);
128  std::streamoff length = staidxFile.tellg();
129  if (length > 0)
130  {
131  staidxFile.seekg (0, staidxFile.beg);
132  staidxFile.read(reinterpret_cast<char*>(m_pStaidxPool), length);
133  }
134  staidxFile.close();
135  }
136 
137  Logger::g_pLogger->LogPrint("Loading Statics: %s\n", staticsFileNameAndPath.c_str());
138 
139  std::ifstream staticsFile;
140  staticsFile.open(staticsFileNameAndPath, std::ios::binary | std::ios::in);
141  if (staticsFile.is_open())
142  {
143  staticsFile.seekg (0, staticsFile.end);
144  std::streamoff length = staticsFile.tellg();
145  if (length > 0)
146  {
147  staticsFile.seekg (0, staticsFile.beg);
148  staticsFile.read(reinterpret_cast<char*>(m_pStaticsPool), length);
149  }
150  m_pStaticsPoolEnd = m_pStaticsPool + length;
151  staticsFile.close();
152  }
153 
154  m_pMapFileStream->open(mapFileNameAndPath, std::ios::out | std::ios::in | std::ios::binary);
155  m_pStaidxFileStream->open(staidxFileNameAndPath, std::ios::out | std::ios::in | std::ios::binary);
156  m_pStaticsFileStream->open(staticsFileNameAndPath, std::ios::out | std::ios::in | std::ios::binary);
157 
158  Logger::g_pLogger->LogPrint("Finished Loading Map!\n");
159 }

176 LPVOID WINAPI FileManager::OnMapViewOfFile(
177  __in HANDLE hFileMappingObject,
178  __in DWORD dwDesiredAccess,
179  __in DWORD dwFileOffsetHigh,
180  __in DWORD dwFileOffsetLow,
181  __in SIZE_T dwNumberOfBytesToMap
182  )
183 {
184  ClientFileHandleSet* pMatchingFileset = NULL;
185 
186  for (std::map<std::string, ClientFileHandleSet*>::const_iterator iterator = m_files.begin(); iterator != m_files.end(); iterator++)
187  {
188  if (iterator->second->m_createFileMappingHandle == hFileMappingObject)
189  {
190  pMatchingFileset = iterator->second;
191  break;
192  }
193  }
194 
195  HANDLE handleToReturn = INVALID_HANDLE_VALUE;
196 
197  if (pMatchingFileset != NULL)
198  {
199  std::string shortFilename = Utils::getFilenameFromPath(pMatchingFileset->m_filename);
200  std::map<std::string, uint32_t>::iterator iterator = m_neededFiles.find(shortFilename);
201 
202  if (iterator != m_neededFiles.end())
203  {
204  if (shortFilename.find("map") != std::string::npos)
205  {
206  handleToReturn = m_pMapPool;
207  }
208  else if (shortFilename.find("statics") != std::string::npos)
209  {
210  handleToReturn = m_pStaticsPool;
211  }
212  else if (shortFilename.find("staidx") != std::string::npos)
213  {
214  handleToReturn = m_pStaidxPool;
215  }
216  }
217  }
218 
219  if (handleToReturn == INVALID_HANDLE_VALUE)
220  {
221  handleToReturn = MapViewOfFile(hFileMappingObject, dwDesiredAccess, dwFileOffsetHigh, dwFileOffsetLow, dwNumberOfBytesToMap);
222  if (pMatchingFileset != NULL)
223  {
224  pMatchingFileset->m_mapFileViewHandle = handleToReturn;
225  }
226  }
227 
228  return handleToReturn;
229 }
