41 /* @brief Self Registering Class Initialize function that sets up packet subscriptions with the network manager
42 */
43 bool LoginHandler::Initialize()
44 {
45  Logger::g_pLogger->LogPrint("Initializing LoginHandler!\n");
46  bool success = true;
47  LoginHandler* pLoginHandler = static_cast<LoginHandler*>(UltimaLive::g_pUltimaLive->Lookup("LoginHandler"));
48 
49  if (pLoginHandler != NULL)
50  {
51  NetworkManager* pManager = static_cast<NetworkManager*>(UltimaLive::g_pUltimaLive->Lookup("NetworkManager"));
52 
53  if (pManager != NULL)
54  {
55  pLoginHandler->m_pManager = pManager;
56  pManager->subscribeToLogout(std::bind(&LoginHandler::onLogoutRequest, pLoginHandler));
57  pManager->subscribeToLoginConfirm(std::bind(&LoginHandler::onLoginConfirm, pLoginHandler, std::placeholders::_1));
58  pManager->subscribeToLoginComplete(std::bind(&LoginHandler::onLoginComplete, pLoginHandler));
59  pManager->subscribeToOnBeforeMapChange(std::bind(&LoginHandler::onBeforeMapChange, pLoginHandler, std::placeholders::_1));
60  pManager->subscribeToServerMobileUpdate(std::bind(&LoginHandler::onServerMobileUpdate, pLoginHandler));
61  pManager->subscribeToMapDefinitionUpdate(std::bind(&LoginHandler::onUpdateMapDefinitions, pLoginHandler, std::placeholders::_1));
62  }
63  else
64  {
65  success = false;
66  Logger::g_pLogger->LogPrintError("LoginHandler: Failed to acquire NetworkManager Instance!\n");
67  }
68  }
69  else
70  {
71  success = false;
72  Logger::g_pLogger->LogPrintError("LoginHandler: Failed to acquire LoginHandler Instance!\n");
73  }
74 
75  return success;
76 }

147 /* @brief Builds a unicode message that can be sent to the client
148  *
149  * @param serial Mobile serial number
150  * @param messageMode 0x00 - Regular, 0x01 - Broadcast, 0x02 - Emote, 0x06 - System, 0x07 - Message, 0x08 - Whisper, 0x09 - Yell
151  * @param hue Color value to display the message
152  * @param pMessage Text content of the message
153  *
154  * @return pointer to the unicode message
155  */
156 uint8_t* BuildUnicodeMessagePacket(uint32_t serial, uint8_t messageMode, uint16_t hue, char* pMessage)
157 {
158  // send welcome message
159  uint8_t aUnicodeMessage[48] = {
160  /* 00 Packet ID */ 0xAE,
161  /* 01 02 Packet Size */ 0x00, 0x00,
162  /* 03 06 Serial */ 0xFF, 0xFF, 0xFF, 0xFF,
163  /* 07 08 Graphic */ 0xFF, 0xFF,
164  /* 09 Mode */ 0x03,
165  /* 10 11 Text Color */ 0x05, 0xb2,
166  /* 12 13 Font */ 0x00, 0x03,
167  /* 14 17 Language */ 0x45, 0x4E, 0x55, 0x00,
168  /* 18 48 Name */ 0x53, 0x79, 0x73, 0x74, 0x65, 0x6D, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
169  /* 18 48 Name */ 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
170 
171 
172  int messageLen = strlen(pMessage);
173  uint16_t packetLength = static_cast<uint16_t>(50 + (messageLen * 2)); // 48 + message + 2 (null terminator)
174  uint8_t* pUnicodePacket = new uint8_t[packetLength];
175 
176  memset(pUnicodePacket, 0x00, packetLength);
177  memcpy(pUnicodePacket, aUnicodeMessage, 48);
178 
179  *reinterpret_cast<uint16_t*>(&pUnicodePacket[1]) = packetLength;
180  *reinterpret_cast<uint32_t*>(&pUnicodePacket[3]) = serial;
181  pUnicodePacket[9] = messageMode;
182  *reinterpret_cast<uint16_t*>(&pUnicodePacket[10]) = hue;
183 
184  for (int i = 0; i < messageLen; ++i)
185  {
186  pUnicodePacket[49 + (2 * i)] = pMessage[i];
187  }
188 
189  return pUnicodePacket;
190 }
