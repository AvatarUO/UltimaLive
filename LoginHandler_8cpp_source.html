164 uint8_t* BuildUnicodeMessagePacket(uint32_t serial, uint8_t messageMode, uint16_t hue, char* pMessage)
165 {
166  // send welcome message
167  uint8_t aUnicodeMessage[48] = {
168  /* 00 Packet ID */ 0xAE,
169  /* 01 02 Packet Size */ 0x00, 0x00,
170  /* 03 06 Serial */ 0xFF, 0xFF, 0xFF, 0xFF,
171  /* 07 08 Graphic */ 0xFF, 0xFF,
172  /* 09 Mode */ 0x03,
173  /* 10 11 Text Color */ 0x05, 0xb2,
174  /* 12 13 Font */ 0x00, 0x03,
175  /* 14 17 Language */ 0x45, 0x4E, 0x55, 0x00,
176  /* 18 48 Name */ 0x53, 0x79, 0x73, 0x74, 0x65, 0x6D, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
177  /* 18 48 Name */ 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
178 
179 
180  int messageLen = strlen(pMessage);
181  uint16_t packetLength = static_cast<uint16_t>(50 + (messageLen * 2)); // 48 + message + 2 (null terminator)
182  uint8_t* pUnicodePacket = new uint8_t[packetLength];
183 
184  memset(pUnicodePacket, 0x00, packetLength);
185  memcpy(pUnicodePacket, aUnicodeMessage, 48);
186 
187  *reinterpret_cast<uint16_t*>(&pUnicodePacket[1]) = packetLength;
188  *reinterpret_cast<uint32_t*>(&pUnicodePacket[3]) = serial;
