37 {
38  unsigned char* pBuffer = static_cast<unsigned char *>(static_cast<unsigned char*>(pData) - searchLen - 1);
39 
40  BOOL found = FALSE;
41  for (unsigned int x = searchLen; x > 0; --x)
42  {
43  for(unsigned int y = 0; y < sigLen; ++y)
44  {
45  if (sigBuffer[y] == 0xCC || pBuffer[x + y] == sigBuffer[y])
