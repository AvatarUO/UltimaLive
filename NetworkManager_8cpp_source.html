46 bool NetworkManager::Initialize()
47 {
48  NetworkManager* pManager = static_cast<NetworkManager*>(UltimaLive::g_pUltimaLive->Lookup("NetworkManager"));
49  bool success = true;
50 
51  if (pManager != NULL)
52  {
53  Client* pClient = static_cast<Client*>(UltimaLive::g_pUltimaLive->Lookup("Client"));
54 
55  if (pClient != NULL)
56  {
57  pManager->m_pClient = pClient;
58 
59  uint32_t versionMajor = UltimaLive::g_pUltimaLive->GetVersionMajor();
60  uint32_t versionMinor = UltimaLive::g_pUltimaLive->GetVersionMinor();
61 
62  Logger::g_pLogger->LogPrint("Initializing Network Manager!\n");
63 
64  pManager->m_sendPacketHandlers = PacketHandlerFactory::GenerateClientPacketHandlers(versionMajor, versionMinor, pManager);
65  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Send-Normal-Packet handlers");
66  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_sendPacketHandlers.size());
67  Logger::g_pLogger->LogPrintTaskStatusResult(true);
68 
69  pManager->m_recvPacketHandlers = PacketHandlerFactory::GenerateServerPacketHandlers(versionMajor, versionMinor, pManager);
70  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Receive-Normal-Packet handlers");
71  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_recvPacketHandlers.size());
72  Logger::g_pLogger->LogPrintTaskStatusResult(true);
73 
74  pManager->m_sendExtendedPacketHandlers = PacketHandlerFactory::GenerateClientExtendedPacketHandlers(versionMajor, versionMinor, pManager);
75  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Send-Extended-Packet handlers");
76  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_sendExtendedPacketHandlers.size());
77  Logger::g_pLogger->LogPrintTaskStatusResult(true);
78 
79  pManager->m_recvExtendedPacketHandlers = PacketHandlerFactory::GenerateServerExtendedPacketHandlers(versionMajor, versionMinor, pManager);
80  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Receive-Extended-Packet handlers");
81  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_recvExtendedPacketHandlers.size());
82  Logger::g_pLogger->LogPrintTaskStatusResult(true);
83 
84  pManager->m_ultimaLiveHandlers = PacketHandlerFactory::GenerateUltimaLiveServerPacketHandlers(versionMajor, versionMinor, pManager);
85  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Receive-UltimaLive-Packet handlers");
86  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_ultimaLiveHandlers.size());
87  Logger::g_pLogger->LogPrintTaskStatusResult(true);
88  }
89  else
90  {
91  Logger::g_pLogger->LogPrintError("NetworkManager: Failed to acquire Client class Instance\n");
92  success = false;
93  }
94  }
95  else
96  {
97  Logger::g_pLogger->LogPrintError("NetworkManager: Failed to acquire NetworkManager Instance\n");
98  success = false;
99  }
100 
101  return success;
102 }

157 bool NetworkManager::OnReceiveServerUltimaLivePacket(unsigned char* pBuffer)
158 {
159  bool retVal = false;
160  uint8_t command = pBuffer[13];
161 
162 #if defined(DEBUG) && defined(PRINT_PACKETS)
163  std::stringstream packetName;
164  packetName << "Received SERVER_ULTIMALIVE_" << ULTIMA_LIVE_PACKET_NAMES[command];
165  packetName << " (0x" << std::hex << (int)command << ")";
166  Logger::g_pLogger->LogPacketToClient("%s\n", packetName.str().c_str());
167 #endif
168 
169  if (m_ultimaLiveHandlers.find(command) != m_ultimaLiveHandlers.end())
170  {
171  m_ultimaLiveHandlers[command]->handlePacket(pBuffer);
172  }
173 
174  return retVal;
175 }

184 bool NetworkManager::OnReceivePacket(unsigned char *pBuffer)
185 {
186  bool retVal = true;
187  uint8_t command = pBuffer[0];
188 
189  switch (command)
190  {
191  case 0x3F:
192  {
193  retVal = OnReceiveServerUltimaLivePacket(pBuffer);
194  }
195  break;
196 
197  case 0xBF:
198  {
199  retVal = OnReceiveExtendedPacket(pBuffer);
200  }
201  break;
202 
203  default:
204  {
205  if (m_recvPacketHandlers.find(command) != m_recvPacketHandlers.end())
206  {
207  retVal = m_recvPacketHandlers[command]->handlePacket(pBuffer);
208  }
209  }
210  break;
211  }
212 
213 #if defined(DEBUG) && defined(PRINT_PACKETS)
214  if (command != 0xBF && command != 0x3F && command != 0x22)
215  {
216  std::stringstream packetName;
217  packetName << "Received SERVER_" << PACKET_NAMES[command];
218  packetName << " (0x" << std::hex << (int)command << ")";
219  Logger::g_pLogger->LogPacketToClient("%s\n", packetName.str().c_str());
220  }
221 #endif
222 
223 #if defined(DEBUG) && defined(PRINT_PACKET_HEX)
224  LocalPeHelper32::HexPrint(pBuffer, 32);
225 #endif
226 
227 
228  return retVal;
229 }

238 bool NetworkManager::OnSendPacket(unsigned char *pBuffer)
239 {
240  bool retVal = true;
241  uint8_t command = pBuffer[0];
242 
243  if (command == 0xBF)
244  {
245  retVal = OnSendExtendedPacket(pBuffer);
246  }
247  else
248  {
249  if (command == 0xF4)
250  {
251  Logger::g_pLogger->LogPrint("CLIENT HAS CRASHED! Crash Report:\n");
252  LocalPeHelper32::HexPrint(pBuffer, *reinterpret_cast<uint16_t*>(&pBuffer[1]));
253  }
254  else if (m_sendPacketHandlers.find(command) != m_sendPacketHandlers.end())
255  {
256  retVal = m_sendPacketHandlers[command]->handlePacket(pBuffer);
257  }
258 
259 #if defined(DEBUG) && defined(PRINT_PACKETS)
260  std::stringstream packetName;
261  packetName << "Sent CLIENT_" << PACKET_NAMES[command];
262  packetName << " (0x" << std::hex << (int)command << ")";
263  Logger::g_pLogger->LogPacketToServer("%s\n", packetName.str().c_str());
264 #endif
265  }
266 
267  return retVal;
268 }

277 bool NetworkManager::OnReceiveExtendedPacket(unsigned char *pBuffer)
278 {
279  bool retVal = true;
280 
281  uint8_t command = static_cast<uint8_t>(ntohs(*reinterpret_cast<uint16_t*>(pBuffer + 3)));
282 
283  if (m_recvExtendedPacketHandlers.find(command) != m_recvExtendedPacketHandlers.end())
284  {
285  retVal = m_recvExtendedPacketHandlers[command]->handlePacket(pBuffer);
286  }
287 
288 #if defined(DEBUG) && defined(PRINT_PACKETS)
289  std::stringstream packetName;
290  packetName << "Received SERVER_EXTENDED_" << EXTENDED_PACKET_NAMES[command];
291  packetName << " (0x" << std::hex << (int)command << ")";
292  Logger::g_pLogger->LogPacketToClient("%s\n", packetName.str().c_str());
293 #endif
294 
295  return retVal;
296 }
