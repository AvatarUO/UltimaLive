39 bool NetworkManager::Initialize()
40 {
41  NetworkManager* pManager = static_cast<NetworkManager*>(UltimaLive::g_pUltimaLive->Lookup("NetworkManager"));
42  bool success = true;
43 
44  if (pManager != NULL)
45  {
46  Client* pClient = static_cast<Client*>(UltimaLive::g_pUltimaLive->Lookup("Client"));
47 
48  if (pClient != NULL)
49  {
50  pManager->m_pClient = pClient;
51 
52  uint32_t versionMajor = UltimaLive::g_pUltimaLive->GetVersionMajor();
53  uint32_t versionMinor = UltimaLive::g_pUltimaLive->GetVersionMinor();
54 
55  Logger::g_pLogger->LogPrint("Initializing Network Manager!\n");
56 
57  pManager->m_sendPacketHandlers = PacketHandlerFactory::GenerateClientPacketHandlers(versionMajor, versionMinor, pManager);
58  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Send-Normal-Packet handlers");
59  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_sendPacketHandlers.size());
60  Logger::g_pLogger->LogPrintTaskStatusResult(true);
61 
62  pManager->m_recvPacketHandlers = PacketHandlerFactory::GenerateServerPacketHandlers(versionMajor, versionMinor, pManager);
63  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Receive-Normal-Packet handlers");
64  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_recvPacketHandlers.size());
65  Logger::g_pLogger->LogPrintTaskStatusResult(true);
66 
67  pManager->m_sendExtendedPacketHandlers = PacketHandlerFactory::GenerateClientExtendedPacketHandlers(versionMajor, versionMinor, pManager);
68  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Send-Extended-Packet handlers");
69  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_sendExtendedPacketHandlers.size());
70  Logger::g_pLogger->LogPrintTaskStatusResult(true);
71 
72  pManager->m_recvExtendedPacketHandlers = PacketHandlerFactory::GenerateServerExtendedPacketHandlers(versionMajor, versionMinor, pManager);
73  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Receive-Extended-Packet handlers");
74  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_recvExtendedPacketHandlers.size());
75  Logger::g_pLogger->LogPrintTaskStatusResult(true);
76 
77  pManager->m_ultimaLiveHandlers = PacketHandlerFactory::GenerateUltimaLiveServerPacketHandlers(versionMajor, versionMinor, pManager);
78  Logger::g_pLogger->LogPrintWithoutDate("%-75s", " Registering Receive-UltimaLive-Packet handlers");
79  Logger::g_pLogger->LogPrintWithoutDate(" Found: %03i ", pManager->m_ultimaLiveHandlers.size());
80  Logger::g_pLogger->LogPrintTaskStatusResult(true);
81  }
82  else
83  {
84  Logger::g_pLogger->LogPrintError("NetworkManager: Failed to acquire Client class Instance\n");
85  success = false;
86  }
87  }
88  else
89  {
90  Logger::g_pLogger->LogPrintError("NetworkManager: Failed to acquire NetworkManager Instance\n");
91  success = false;
92  }
93 
94  return success;
95 }

132 bool NetworkManager::OnReceiveServerUltimaLivePacket(unsigned char* pBuffer)
133 {
134  bool retVal = false;
135  uint8_t command = pBuffer[13];
136 
137 #if defined(DEBUG) && defined(PRINT_PACKETS)
138  std::stringstream packetName;
139  packetName << "Received SERVER_ULTIMALIVE_" << ULTIMA_LIVE_PACKET_NAMES[command];
140  packetName << " (0x" << std::hex << (int)command << ")";
141  Logger::g_pLogger->LogPacketToClient("%s\n", packetName.str().c_str());
142 #endif
143 
144  if (m_ultimaLiveHandlers.find(command) != m_ultimaLiveHandlers.end())
145  {
146  m_ultimaLiveHandlers[command]->handlePacket(pBuffer);
147  }
148 
149  return retVal;
150 }

152 bool NetworkManager::OnReceivePacket(unsigned char *pBuffer)
153 {
154  bool retVal = true;
155  uint8_t command = pBuffer[0];
156 
157  switch (command)
158  {
159  case 0x3F:
160  {
161  retVal = OnReceiveServerUltimaLivePacket(pBuffer);
162  }
163  break;
164 
165  case 0xBF:
166  {
167  retVal = OnReceiveExtendedPacket(pBuffer);
168  }
169  break;
170 
171  default:
172  {
173  if (m_recvPacketHandlers.find(command) != m_recvPacketHandlers.end())
174  {
175  retVal = m_recvPacketHandlers[command]->handlePacket(pBuffer);
176  }
177  }
178  break;
179  }
180 
181 #if defined(DEBUG) && defined(PRINT_PACKETS)
182  if (command != 0xBF && command != 0x3F && command != 0x22)
183  {
184  std::stringstream packetName;
185  packetName << "Received SERVER_" << PACKET_NAMES[command];
186  packetName << " (0x" << std::hex << (int)command << ")";
187  Logger::g_pLogger->LogPacketToClient("%s\n", packetName.str().c_str());
188  }
189 #endif
190 
191 #if defined(DEBUG) && defined(PRINT_PACKET_HEX)
192  LocalPeHelper32::HexPrint(pBuffer, 32);
193 #endif
194 
195 
196  return retVal;
197 }

199 bool NetworkManager::OnSendPacket(unsigned char *pBuffer)
200 {
201  bool retVal = true;
202  uint8_t command = pBuffer[0];
203 
204  if (command == 0xBF)
205  {
206  retVal = OnSendExtendedPacket(pBuffer);
207  }
208  else
209  {
210  if (command == 0xF4)
211  {
212  Logger::g_pLogger->LogPrint("CLIENT HAS CRASHED! Crash Report:\n");
213  LocalPeHelper32::HexPrint(pBuffer, *reinterpret_cast<uint16_t*>(&pBuffer[1]));
214  }
215  else if (m_sendPacketHandlers.find(command) != m_sendPacketHandlers.end())
216  {
217  retVal = m_sendPacketHandlers[command]->handlePacket(pBuffer);
218  }
219 
220 #if defined(DEBUG) && defined(PRINT_PACKETS)
221  std::stringstream packetName;
222  packetName << "Sent CLIENT_" << PACKET_NAMES[command];
223  packetName << " (0x" << std::hex << (int)command << ")";
224  Logger::g_pLogger->LogPacketToServer("%s\n", packetName.str().c_str());
225 #endif
226  }
227 
228  return retVal;
229 }

231 bool NetworkManager::OnReceiveExtendedPacket(unsigned char *pBuffer)
232 {
233  bool retVal = true;
234 
235  uint8_t command = static_cast<uint8_t>(ntohs(*reinterpret_cast<uint16_t*>(pBuffer + 3)));
236 
237  if (m_recvExtendedPacketHandlers.find(command) != m_recvExtendedPacketHandlers.end())
238  {
239  retVal = m_recvExtendedPacketHandlers[command]->handlePacket(pBuffer);
240  }
241 
242 #if defined(DEBUG) && defined(PRINT_PACKETS)
243  std::stringstream packetName;
244  packetName << "Received SERVER_EXTENDED_" << EXTENDED_PACKET_NAMES[command];
245  packetName << " (0x" << std::hex << (int)command << ")";
246  Logger::g_pLogger->LogPacketToClient("%s\n", packetName.str().c_str());
247 #endif
248 
249  return retVal;
250 }
