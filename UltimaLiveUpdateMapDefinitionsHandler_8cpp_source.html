46 bool UltimaLiveUpdateMapDefinitionsHandler::handlePacket(uint8_t* pPacketData)
47 {
48  std::vector<MapDefinition> definitions;
49 
50  uint32_t count = ntohl(*reinterpret_cast<uint32_t*>(&pPacketData[7]));
51  int newNumMaps = (count * 7) / 9;
52 
53  for (int i = 0; i < newNumMaps; i++)
54  {
55  int offset = 15 + (i * 9);
56  uint8_t mapNumber = (uint8_t)pPacketData[offset]; //iteration byte 000 - map file index number
57  uint16_t width = ntohs(*reinterpret_cast<uint16_t*>(&pPacketData[offset + 1])); //iteration byte 001 to 002 - map width
58  uint16_t height = ntohs(*reinterpret_cast<uint16_t*>(&pPacketData[offset + 3])); //iteration byte 003 to 004 - map height
59  uint16_t wrapX = ntohs(*reinterpret_cast<uint16_t*>(&pPacketData[offset + 5])); //iteration byte 005 to 006 - wrap around dimension X
60  uint16_t wrapY = ntohs(*reinterpret_cast<uint16_t*>(&pPacketData[offset + 7])); //iteration byte 007 to 008 - wrap around dimension Y
61 
62  if ((width > 0) && (height > 0) && (wrapX > 0) && (wrapY > 0))
63  {
64  MapDefinition def;
65  def.mapNumber = mapNumber;
66  def.mapWidthInTiles = width;
67  def.mapHeightInTiles = height;
68  def.mapWrapWidthInTiles = wrapX;
69  def.mapWrapHeightInTiles = wrapY;
70  definitions.push_back(def);
71  }
72  else
73  {
74  Logger::g_pLogger->LogPrint("Received bad definition for map %i\n", i);
75  }
76  }
77 
78  m_pManager->onMapDefinitionUpdate(definitions);
79  return false;
80 }
