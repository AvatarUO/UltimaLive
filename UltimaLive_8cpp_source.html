34 UltimaLive::UltimaLive()
35  : m_pClient(NULL),
36  m_versionMajor(0),
37  m_versionMinor(0),
38  m_registrations()
39 {
40  std::string currentModuleFullPath = Utils::GetCurrentModuleFullPath();
41  uint32_t fileVersionSize = GetFileVersionInfoSizeA(currentModuleFullPath.c_str(), NULL);
42  if (fileVersionSize > 0)
43  {
44  char* pVersionBuffer = new char[fileVersionSize];
45  char* pQueryValueBuffer = NULL;
46 
47  if (GetFileVersionInfo(currentModuleFullPath.c_str(), NULL, fileVersionSize, pVersionBuffer))
48  {
49  uint32_t size = 0;
50  if (VerQueryValue(pVersionBuffer, "\\", reinterpret_cast<VOID FAR* FAR*>(&pQueryValueBuffer), &size))
51  {
52  if (size > 0)
53  {
54  VS_FIXEDFILEINFO *verInfo = (VS_FIXEDFILEINFO *)pQueryValueBuffer;
55  if (verInfo->dwSignature == 0xfeef04bd)
56  {
57  int major = HIWORD(verInfo->dwFileVersionMS);
58  int minor = LOWORD(verInfo->dwFileVersionMS);
59  int buildMajor = HIWORD(verInfo->dwFileVersionLS);
60  int buildMinor = LOWORD(verInfo->dwFileVersionLS);
61 
62  char buffer[20];
63  memset(buffer, 0x00, 20);
64  sprintf_s(buffer, "%i.%i.%i.%i", major, minor, buildMajor, buildMinor);
65 
66  m_versionMajor = verInfo->dwFileVersionMS;
67  m_versionMinor = verInfo->dwFileVersionLS;
68  }
69  }
70  }
71  }
72 
73  delete pVersionBuffer;
74  }
75 }

100  }
101 
102  if (safeToProceed)
103  {
104  if (!ClientRedirections::InstallClientHooks())
105  {
106  #ifdef DEBUG
107  system("pause");
108  #else
109  MessageBoxA(NULL,"Unable to install client hooks, exiting.", "Error", NULL);
110  #endif
111  exit(-1);
112  }
113 
114  #if defined DEBUG && defined DOTNET
115  Logger::g_pLogger->LogPrint("Launching .NET");
116  g_pDotNetHost = new DotNetHost();
117  g_pDotNetHost->LaunchCLR();
118  uint32_t result = g_pDotNetHost->LaunchNetAssembly("UltimaLiveObjectViewer.dll", "UltimaLiveObjectViewer.ObjectViewer", "ShowMainWindow", "asdf");
119  #endif
120  }
121  else // we failed to find all of the required client internal addresses
122  {
123  Logger::g_pLogger->g_pLogger->LogPrintWarning("Reverting to regular client.\n");
124  LocalPeHelper32 ultimaHelper(Utils::GetCurrentModuleName());
125 
126  bool success = true;

129  {
130  success = false;
131  }
132 
133  if (success)

136  {
137  success = false;
138  }
139  }
140 
141  if (!success)
142  {
143 
144  Logger::g_pLogger->LogPrintError("Could not restore CreateFileA!\n");
145  Logger::g_pLogger->LogPrintError("Cannot recover! Exiting!\n");
146  #ifdef DEBUG

150  #endif
